6  module subroutine iii_initialize(this, err)
7  ! Arguments
8  class(infinite_interval_integrator), intent(inout) :: this
9  class(errors), intent(inout), optional, target :: err
10 
11  ! Local Variables
12  integer(int32) :: flag, lwork, liwork
13  class(errors), pointer :: errmgr
14  type(errors), target :: deferr
15 
16  ! Set up the error handling
17  if (present(err)) then
18  errmgr => err
19  else
20  errmgr => deferr
21  end if
22 
23  ! Initialization
24  liwork = this%get_max_subintervals()
25  lwork = 4 * this%get_max_subintervals()
26 
27  ! Memory Allocation
28  if (allocated(this%m_work)) deallocate(this%m_work)
29  if (allocated(this%m_iwork)) deallocate(this%m_iwork)
30  allocate(this%m_work(lwork), stat = flag)
31  if (flag == 0) allocate(this%m_iwork(liwork), stat = flag)
32  if (flag /= 0) then
33  call errmgr%report_error("iii_initialize", &
34  "Insufficient memory available.", int_out_of_memory_error)
35  return
36  end if
37  end subroutine

40  module function iii_integrate(this, fcn, bound, upper, info, err) result(rst)
41  ! Arguments
42  class(infinite_interval_integrator), intent(inout) :: this
43  procedure(integrand), intent(in), pointer :: fcn
44  real(real64), intent(in), optional :: bound
45  logical, intent(in), optional :: upper
46  type(integration_behavior), intent(out), optional :: info
47  class(errors), intent(inout), optional, target :: err
48  real(real64) :: rst
49 
50  ! Local Variables
51  integer(int32) :: inf, neval, ier, limit, lwork, last
52  real(real64) :: bnd, epsabs, epsrel, abserr
53  class(errors), pointer :: errmgr
54  type(errors), target :: deferr
55  character(len = 256) :: errmsg
56 
57  ! Error Handling Set-Up
58  if (present(err)) then
59  errmgr => err
60  else
61  errmgr => deferr
62  end if
63 
64  ! Initialization
65  limit = this%get_max_subintervals()
66  lwork = 4 * this%get_max_subintervals()
67  bnd = 0.0d0
68  inf = 2
69  if (present(bound)) then
70  bnd = bound
71  if (present(upper)) then
72  if (upper) then
73  ! bound is the upper limit
74  inf = -1
75  else
76  ! bound is the lower limit
77  inf = 1
78  end if
79  else
80  ! ERROR: upper must be supplied if bound is supplied
81  call errmgr%report_error("iii_integrate", &
82  "The parameter upper must be supplied when the " // &
83  "parameter bound is supplied.", &
84  int_invalid_operation_error)
85  return
86  end if
87  end if
88 
89  ! Ensure the object is initialized properly
90  if (.not.allocated(this%m_work) .or. .not.allocated(this%m_iwork)) then
91  call this%initialize(errmgr)
92  if (errmgr%has_error_occurred()) return
93  end if
94  if (size(this%m_work) < lwork) then
95  call this%initialize(errmgr)
96  if (errmgr%has_error_occurred()) return
97  end if
98 
99  ! Get the tolerance info
100  epsrel = this%get_rel_tol()
101  epsabs = this%get_abs_tol()
102 
103  ! Compute the integral
104  call dqagi(local_fcn, bnd, inf, epsabs, epsrel, rst, abserr, neval, &
105  ier, limit, lwork, last, this%m_iwork, this%m_work)
106 
107  ! Collect info regarding the integration process
108  if (present(info)) then
109  info%error_estimate = abserr
110  info%evaluation_count = neval
111  info%subinterval_count = last
112  end if
113 
114  ! Check for errors
115  select case (ier)
116  case (1)
117  write(errmsg, '(AI0A)') "The maximum number (", limit, &
118  ") of subdivisions allowed has been achieved."
119  call errmgr%report_error("iii_integrate", trim(errmsg), &
120  int_count_exceeded_error)
121  return
122  case (2)
123  call errmgr%report_error("iii_integrate", &
124  "The occurence of roundoff error has been detected " // &
125  "thereby preventing the requested tolerance from being " // &
126  "achieved.", int_round_off_error)
127  return
128  case (3)
129  call errmgr%report_error("iii_integrate", &
130  "The integrand appears to be behaving too poorly to proceed.", &
131  int_integrand_behavior_error)
132  return
133  case (4)
134  call errmgr%report_error("iii_integrate", &
135  "The algorithm cannot converge with the assigned tolerances.", &
136  int_convergence_error)
137  return
138  case (5)
139  call errmgr%report_error("iii_integrate", &
140  "The integral is likely divergent", int_divergent_error)
141  return
142  case (6)
143  call errmgr%report_error("iii_integrate", &
144  "An invalid input has been provided.", int_invalid_input_error)
145  return
146  end select
